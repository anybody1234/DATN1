\begin{longtable}{p{3cm} p{11cm}}
	\hline
   \textbf{Thuật ngữ}  & \textbf{Diễn giải} \\ \hline
	\textbf{Browser} & Trình duyệt \\
	\textbf{Cache} & Bộ nhớ đệm, lưu tạm dữ liệu để truy xuất nhanh hơn \\
	\textbf{Token} & Mã thông báo dùng để xác thực hoặc định danh một phiên làm việc \\
	\textbf{Endpoint} & Một địa chỉ (đường dẫn) cụ thể mà API cung cấp để client gọi tới \\
	\textbf{Middleware} & Tầng trung gian xử lý request/response trước khi tới tầng nghiệp vụ \\
	\textbf{Callback} & Hàm được gọi lại sau khi một thao tác (thường là bất đồng bộ) hoàn tất \\
	\textbf{Webhook} & Cơ chế một hệ thống bên ngoài gọi ngược lại API của hệ thống khi có sự kiện xảy ra \\
	\textbf{Sandbox} & Môi trường thử nghiệm độc lập, tách biệt với môi trường thực tế \\
	\textbf{Hook} (React) & Hàm đặc biệt cho phép component sử dụng state và các tính năng khác của React \\
	\textbf{Component} & Đơn vị giao diện độc lập, có thể tái sử dụng trong ứng dụng frontend \\
    \hline
\end{longtable}
